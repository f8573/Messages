```sql
);
```

### Table Modification
```sql
ALTER TABLE miniapp_events ADD COLUMN event_type miniapp_event_type NOT NULL DEFAULT 'storage_updated';
```

### Append-Only Trigger
```sql
CREATE FUNCTION enforce_miniapp_events_append_only() RETURNS TRIGGER AS $$
BEGIN
  IF TG_OP = 'UPDATE' THEN
    RAISE EXCEPTION 'miniapp_events table is append-only: UPDATE not allowed';
  END IF;
  IF TG_OP = 'DELETE' THEN
    RAISE EXCEPTION 'miniapp_events table is append-only: DELETE not allowed';
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER miniapp_events_append_only_trigger
BEFORE UPDATE OR DELETE ON miniapp_events
FOR EACH ROW
EXECUTE FUNCTION enforce_miniapp_events_append_only();
```

### Performance Indices

```sql
CREATE INDEX idx_miniapp_events_type ON miniapp_events(app_session_id, event_type);
CREATE INDEX idx_miniapp_events_actor ON miniapp_events(actor_user_id, created_at);
CREATE INDEX idx_miniapp_events_created ON miniapp_events(created_at);
```

## API SPECIFICATION

### New Endpoint: Get Session Events

**Route:** `GET /v1/sessions/{id}/events`

**Authentication:** Required (Bearer token)

**Path Parameters:**
- `id` (UUID): Session ID

**Query Parameters:**
- `event_type` (string, optional): Filter by event type (session_created, session_joined, storage_updated, snapshot_written, message_projected)
- `limit` (integer, optional, default 100): Max events per page (1-1000)
- `offset` (integer, optional, default 0): Pagination offset

**Response (200 OK):**
```json
{
  "events": [
    {
      "event_seq": 1,
      "event_type": "session_created",
      "actor_id": "alice-uuid",
      "body": {
        "participant_count": 1,
        "permissions": ["storage.read", "storage.write"],
        "participants": [...]
      },
      "created_at": "2026-03-21T10:30:45Z"
    }
  ],
  "limit": 100,
  "offset": 0
}
```

**Status Codes:**
- `200 OK`: Events retrieved successfully
- `401 Unauthorized`: Missing or invalid auth token
- `403 Forbidden`: User doesn't have access to session
- `404 Not Found`: Session not found
- `500 Internal Server Error`: Database or server error

**Example Requests:**
```bash
# Get all events for a session
curl -H "Authorization: Bearer $TOKEN" \
  https://api.ohmf.dev/v1/sessions/123e4567-e89b-12d3-a456-426614174000/events

# Filter by event type
curl -H "Authorization: Bearer $TOKEN" \
  'https://api.ohmf.dev/v1/sessions/123e4567-e89b-12d3-a456-426614174000/events?event_type=session_joined'

# Paginate through events
curl -H "Authorization: Bearer $TOKEN" \
  'https://api.ohmf.dev/v1/sessions/123e4567-e89b-12d3-a456-426614174000/events?limit=10&offset=0'
```
